\section{Theoretical Gaps}
\label{sec:theoretical_gaps}

\subsection{Finite-Key Security Analysis}

\textbf{Gap:} Nguyen et al. uses asymptotic security analysis. Practical implementations require finite-key bounds.

\textbf{Required Work:}
\begin{enumerate}
    \item Security bounds for practical key lengths ($10^6$--$10^9$ bits)
    \item Minimum block size determination for target security
    \item Composable security proof incorporating retransmission
\end{enumerate}

\textbf{Related Literature:}
\begin{itemize}
    \item Tomamichel et al. (2012): Finite-key framework
    \item arXiv:2501.10278 (2025): Finite-key for imperfect heterodyne
\end{itemize}

\subsection{Advanced Eavesdropper Models}

\textbf{Gap:} Only Unauthorized Receiver Attack (URA) analyzed. More sophisticated attacks not considered.

\textbf{Required Work:}
\begin{enumerate}
    \item \textbf{Collective Attacks:}
    \begin{itemize}
        \item Eve performs identical operation on each signal
        \item Stores quantum memory for later measurement
    \end{itemize}

    \item \textbf{Coherent Attacks:}
    \begin{itemize}
        \item Most general attack strategy
        \item Joint operation on entire transmission
    \end{itemize}

    \item \textbf{Side-Channel Attacks:}
    \begin{itemize}
        \item Timing information leakage
        \item Modulator imperfections
        \item Detector vulnerabilities
    \end{itemize}
\end{enumerate}

\subsection{Hybrid ARQ-FEC Analysis}

\textbf{Gap:} Pure ARQ approach may not be optimal for all conditions.

\textbf{Research Questions:}
\begin{itemize}
    \item When does hybrid ARQ+FEC outperform pure ARQ?
    \item Optimal FEC code rate for retransmission scenarios
    \item Security implications of hybrid approaches
\end{itemize}

\section{Practical Implementation Gaps}
\label{sec:practical_gaps}

\subsection{Pointing and Tracking}

\textbf{Gap:} Perfect beam tracking assumed. Realistic pointing errors not modeled.

\textbf{Required Work:}
\begin{enumerate}
    \item \textbf{Pointing Error Model:}
    \begin{equation}
    h_p = A_0 \exp\left(-\frac{2r_p^2}{\omega_{eq}^2}\right)
    \label{eq:pointing_error}
    \end{equation}
    where $r_p$ is pointing jitter radius

    \item \textbf{Acquisition Protocol:}
    \begin{itemize}
        \item Initial beam acquisition time
        \item Tracking loop bandwidth requirements
        \item Re-acquisition after interruption
    \end{itemize}

    \item \textbf{Combined Effects:}
    \begin{itemize}
        \item Pointing + turbulence interaction
        \item Impact on QBER and $P_{sift}$
        \item Adaptation of DT coefficient
    \end{itemize}
\end{enumerate}

\subsection{Phase Noise and Synchronization}

\textbf{Gap:} Ideal local oscillator and perfect phase synchronization assumed.

\textbf{Challenges:}
\begin{itemize}
    \item LO phase noise impact on QPSK detection
    \item Doppler shift compensation for LEO satellite
    \item Carrier frequency offset estimation
\end{itemize}

\subsection{Experimental Validation}

\textbf{Gap:} All results are simulation-based.

\textbf{Validation Pathway:}
\begin{enumerate}
    \item \textbf{Component Level:}
    \begin{itemize}
        \item QPSK modulator characterization
        \item DT/HD receiver implementation
        \item APD performance verification
    \end{itemize}

    \item \textbf{Subsystem Level:}
    \begin{itemize}
        \item End-to-end link demonstration
        \item ARQ protocol implementation
        \item Buffer management testing
    \end{itemize}

    \item \textbf{System Level:}
    \begin{itemize}
        \item Ground testbed with emulated satellite channel
        \item Turbulence chamber testing
        \item Field trials (ground-to-ground)
    \end{itemize}
\end{enumerate}

\section{System-Level Gaps}
\label{sec:system_gaps}

\subsection{LEO Constellation Integration}

\textbf{Gap:} Single satellite link analyzed. Constellation operation not considered.

\textbf{Research Directions:}
\begin{enumerate}
    \item \textbf{Multi-Satellite Coverage:}
    \begin{itemize}
        \item Optimal constellation design for Vietnam
        \item Coverage overlap analysis
        \item Handover frequency estimation
    \end{itemize}

    \item \textbf{Handover Protocols:}
    \begin{itemize}
        \item Key continuity during handover
        \item Buffer management across satellites
        \item Retransmission state transfer
    \end{itemize}

    \item \textbf{Key Routing:}
    \begin{itemize}
        \item Inter-satellite key relay
        \item Trusted node requirements
        \item Network key rate optimization
    \end{itemize}
\end{enumerate}

\subsection{Hybrid FSO/RF Architecture}

\textbf{Gap:} Pure FSO system assumed. Backup RF channel not integrated.

\textbf{Opportunities:}
\begin{itemize}
    \item RF backup during weather outages
    \item Classical channel for ACK/NACK (already assumed)
    \item Hybrid key management protocols
\end{itemize}

\subsection{Network Integration}

\textbf{Gap:} Stand-alone QKD system. Integration with existing networks not addressed.

\textbf{Considerations:}
\begin{itemize}
    \item Key management system integration
    \item Classical encryption interoperability
    \item Network protocol stack placement
\end{itemize}

\section{Future Research Directions}
\label{sec:future_directions}

\subsection{Near-Term (1--2 Years)}

\begin{enumerate}
    \item \textbf{Finite-Key Analysis Extension}
    \begin{itemize}
        \item Incorporate finite-size corrections into QBER analysis
        \item Determine minimum key length for target security level
        \item Optimize block size for satellite pass duration
    \end{itemize}

    \item \textbf{Pointing Error Integration}
    \begin{itemize}
        \item Add realistic pointing jitter model
        \item Analyze combined pointing and turbulence effects
        \item Develop adaptive DT coefficient adjustment
    \end{itemize}

    \item \textbf{Ground Testbed Development}
    \begin{itemize}
        \item Implement QPSK modulator and DT/HD receiver
        \item Validate with emulated satellite channel
        \item Demonstrate ARQ protocol operation
    \end{itemize}
\end{enumerate}

\subsection{Medium-Term (3--5 Years)}

\begin{enumerate}
    \item \textbf{Machine Learning Integration}
    \begin{itemize}
        \item Channel state prediction for proactive parameter adjustment
        \item Optimal DT coefficient selection via reinforcement learning
        \item Anomaly detection for security monitoring
    \end{itemize}

    \item \textbf{LEO Constellation Analysis}
    \begin{itemize}
        \item Multi-satellite coverage optimization for Vietnam
        \item Handover protocol development
        \item Key routing algorithm design
    \end{itemize}

    \item \textbf{Tropical Atmosphere Modeling}
    \begin{itemize}
        \item Vietnam-specific turbulence profiles
        \item Monsoon season characterization
        \item Optimal ground station site selection
    \end{itemize}
\end{enumerate}

\subsection{Long-Term (5+ Years)}

\begin{enumerate}
    \item \textbf{Vietnamese Quantum Satellite Mission}
    \begin{itemize}
        \item Payload design based on PTIT research
        \item Ground station network development
        \item International collaboration framework
    \end{itemize}

    \item \textbf{Regional Quantum Network}
    \begin{itemize}
        \item ASEAN quantum connectivity
        \item Cross-border secure communication
        \item Regional standards development
    \end{itemize}

    \item \textbf{Quantum Internet Integration}
    \begin{itemize}
        \item Entanglement-based protocols
        \item Quantum repeater integration
        \item Global quantum network participation
    \end{itemize}
\end{enumerate}
