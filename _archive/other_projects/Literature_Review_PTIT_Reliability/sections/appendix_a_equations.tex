\section{Channel Model Equations}
\label{app:channel_eqs}

\subsection{Combined Channel Coefficient}

\begin{equation}
h = h_l \cdot h_a \cdot h_s \cdot h_t
\tag{A.1}
\end{equation}

\subsection{Free-Space Path Loss}

\begin{equation}
h_l = \left( \frac{\lambda}{4\pi D_{SG}} \right)^2
\tag{A.2}
\end{equation}

\subsection{Slant Range}

\begin{equation}
D_{SG} = \sqrt{(H_S - H_G)^2 \sec^2(\zeta) + 2R_E(H_S - H_G)\sec(\zeta)}
\tag{A.3}
\end{equation}

\subsection{Atmospheric Attenuation}

\begin{equation}
h_a = \exp\left( -\gamma \cdot \frac{H_\beta - H_G}{\cos(\zeta)} \right)
\tag{A.4}
\end{equation}

\subsection{Hufnagel-Valley Turbulence Profile}

\begin{equation}
C_n^2(h) = 0.00594 \left(\frac{w}{27}\right)^2 (10^{-5}h)^{10} e^{-h/1000} + 2.7 \times 10^{-16} e^{-h/1500} + C_n^2(0) e^{-h/100}
\tag{A.5}
\end{equation}

\subsection{Rytov Variance}

\begin{equation}
\sigma_R^2 = 2.25 k^{7/6} \sec^{11/6}(\zeta) \int_{H_G}^{H_S} C_n^2(h) \left(1 - \frac{h-H_G}{H_S-H_G}\right)^{5/6} (h-H_G)^{5/6} dh
\tag{A.6}
\end{equation}

\subsection{Gamma-Gamma Distribution}

\begin{equation}
f_{h_t}(h_t) = \frac{2(\alpha\beta)^{(\alpha+\beta)/2}}{\Gamma(\alpha)\Gamma(\beta)} h_t^{(\alpha+\beta)/2-1} K_{\alpha-\beta}\left(2\sqrt{\alpha\beta h_t}\right)
\tag{A.7}
\end{equation}

\subsection{Scintillation Parameters}

\begin{align}
\alpha &= \left[ \exp\left( \frac{0.49\sigma_R^2}{(1 + 1.11\sigma_R^{12/5})^{7/6}} \right) - 1 \right]^{-1} \tag{A.8a}\\
\beta &= \left[ \exp\left( \frac{0.51\sigma_R^2}{(1 + 0.69\sigma_R^{12/5})^{5/6}} \right) - 1 \right]^{-1} \tag{A.8b}
\end{align}

\section{Detection Equations}
\label{app:detection_eqs}

\subsection{QPSK Phase States}

\begin{equation}
\phi_A \in \left\{ \frac{\pi}{4}, \frac{3\pi}{4}, \frac{5\pi}{4}, -\frac{\pi}{4} \right\}
\tag{A.9}
\end{equation}

\subsection{Dual-Threshold Decision}

\begin{equation}
\text{Decision} = \begin{cases}
0 & \text{if } i \geq d_0 \\
1 & \text{if } i \leq d_1 \\
X & \text{otherwise (erasure)}
\end{cases}
\tag{A.10}
\end{equation}

\subsection{Threshold Configuration}

\begin{align}
d_0 &= d + \varsigma \cdot \sigma \tag{A.11a}\\
d_1 &= d - \varsigma \cdot \sigma \tag{A.11b}
\end{align}

\subsection{Received Power}

\begin{equation}
P_R = P_T \cdot G_T \cdot G_R \cdot h
\tag{A.12}
\end{equation}

\subsection{SNR for Heterodyne Detection}

\begin{equation}
\text{SNR} = \frac{(\Re P_R \bar{g})^2}{\sigma_{shot}^2 + \sigma_{thermal}^2}
\tag{A.13}
\end{equation}

\subsection{Noise Variances}

\begin{align}
\sigma_{shot}^2 &= 2q(\Re P_{LO} + I_d)B \cdot \bar{g}^2 F \tag{A.14a}\\
\sigma_{thermal}^2 &= \frac{4k_B T B}{R_L} \tag{A.14b}
\end{align}

\section{QBER Equations}
\label{app:qber_eqs}

\subsection{Bit Error Probability}

\begin{equation}
P_e = \int_0^\infty Q\left(\sqrt{\frac{2h \cdot \text{SNR}}{1 + h \cdot \text{SNR}}}\right) f_{h_t}(h) dh
\tag{A.15}
\end{equation}

\subsection{Q-Function}

\begin{equation}
Q(x) = \frac{1}{\sqrt{2\pi}} \int_x^\infty e^{-t^2/2} dt
\tag{A.16}
\end{equation}

\subsection{Conditional QBER}

\begin{equation}
\text{QBER} = \frac{P_e}{P_{sift}}
\tag{A.17}
\end{equation}

\subsection{Sifting Probability}

\begin{equation}
P_{sift} = P(|i - d| > \varsigma \sigma) = 1 - P_{erasure}
\tag{A.18}
\end{equation}

\section{Markov Chain Model Equations}
\label{app:markov_eqs}

\subsection{State Space Definition}

\begin{equation}
\text{State: } (n, s, m) \text{ where } n \in [0, C], s \in \{B, G\}, m \in [1, M]
\tag{A.19}
\end{equation}

\subsection{Channel Transition Probabilities}

\begin{align}
P(G \rightarrow B) &= p_{GB} \tag{A.20a}\\
P(B \rightarrow G) &= p_{BG} \tag{A.20b}
\end{align}

\subsection{Steady-State Distribution}

\begin{equation}
\boldsymbol{\pi} = \boldsymbol{\pi} \mathbf{P}
\tag{A.21}
\end{equation}

\subsection{Key Loss Rate}

\begin{equation}
\text{KLR} = \sum_{s \in \{B,G\}} \pi_{C,s,M} \cdot P_{loss}^s
\tag{A.22}
\end{equation}

\section{Security Equations}
\label{app:security_eqs}

\subsection{Eve's Received Power}

\begin{equation}
P_{R,Eve} = P_T \cdot G_T \cdot G_R \cdot h \cdot A_0 \exp\left(-\frac{2D_{E-B}^2}{\omega_{Deq}^2}\right)
\tag{A.23}
\end{equation}

\subsection{Secure Key Rate (Asymptotic)}

\begin{equation}
R_{secure} = R_{sift} \cdot [1 - H(\text{QBER}) - f \cdot H(\text{QBER})]
\tag{A.24}
\end{equation}

\subsection{Binary Entropy Function}

\begin{equation}
H(p) = -p \log_2(p) - (1-p) \log_2(1-p)
\tag{A.25}
\end{equation}
